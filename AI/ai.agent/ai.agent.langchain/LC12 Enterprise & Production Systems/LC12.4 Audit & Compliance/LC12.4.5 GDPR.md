# GDPR Compliance

```python
class GDPRService:
    async def delete_user_data(self, user_id: str):
        # Right to be forgotten
        await user_service.anonymize(user_id)
        await conversation_service.delete_by_user(user_id)
        await vectorstore.delete(filter={"user_id": user_id})
        
        audit_log.record("gdpr_deletion", user_id, "all_data", {})
    
    async def export_user_data(self, user_id: str) -> dict:
        # Right to data portability
        return {
            "user": await user_service.get(user_id),
            "conversations": await conversation_service.get_all(user_id),
            "preferences": await preferences_service.get(user_id)
        }
    
    def check_consent(self, user_id: str, purpose: str) -> bool:
        consent = consent_db.get(user_id, purpose)
        return consent and consent["granted"]
```
