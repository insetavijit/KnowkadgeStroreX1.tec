# Compliance Reporting

```python
class ComplianceReporter:
    async def generate_report(self, period_start: date, period_end: date) -> dict:
        return {
            "period": f"{period_start} to {period_end}",
            "audit_log_count": await audit_log.count(period_start, period_end),
            "data_breaches": await security_incidents.count(period_start, period_end),
            "gdpr_requests": {
                "deletions": await gdpr_service.count_deletions(period_start, period_end),
                "exports": await gdpr_service.count_exports(period_start, period_end)
            },
            "access_reviews": await access_review.get_completed(period_start, period_end),
            "compliance_checks": await compliance_monitor.get_history(period_start, period_end)
        }
    
    async def export_pdf(self, report: dict) -> bytes:
        return pdf_generator.generate(report)
```
