# Compliance Tracking

```python
class ComplianceMonitor:
    CHECKS = {
        "encryption_at_rest": lambda: check_encryption(),
        "mfa_enabled": lambda: check_mfa_adoption(),
        "data_retention": lambda: check_retention_policy(),
        "audit_logging": lambda: check_audit_logs(),
    }
    
    async def run_checks(self) -> dict:
        results = {}
        for check_name, check_func in self.CHECKS.items():
            try:
                results[check_name] = {
                    "status": "pass" if check_func() else "fail",
                    "checked_at": datetime.utcnow()
                }
            except Exception as e:
                results[check_name] = {"status": "error", "error": str(e)}
        
        return results
```
