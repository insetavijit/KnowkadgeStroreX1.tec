# LDAP Authentication
import ldap3

def authenticate_ldap(username: str, password: str) -> bool:
    server = ldap3.Server('ldap://ldap.company.com')
    conn = ldap3.Connection(server, f'cn={username},dc=company,dc=com', password)
    return conn.bind()

# Active Directory Integration
from ldap3 import Server, Connection, ALL

def get_user_groups(username: str) -> list:
    server = Server('ldap://ad.company.com', get_info=ALL)
    conn = Connection(server, 'domain\\admin', 'password', auto_bind=True)
    
    conn.search(
        'dc=company,dc=com',
        f'(sAMAccountName={username})',
        attributes=['memberOf']
    )
    
    return conn.entries[0].memberOf.values if conn.entries else []
