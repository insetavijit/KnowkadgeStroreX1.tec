            "description": description,
            "created_at": datetime.utcnow(),
            "status": "open",
            "assigned_to": await self.auto_assign(severity)
        }
        
        await incident_db.insert(incident)
        await self.notify_oncall(incident)
        
        return incident
    
    async def escalate(self, incident_id: str):
        incident = await incident_db.get(incident_id)
        escalation_chain = ["engineer", "lead", "manager", "director"]
        
        current_level = escalation_chain.index(incident["assigned_to"])
        if current_level < len(escalation_chain) - 1:
            incident["assigned_to"] = escalation_chain[current_level + 1]
            await self.notify_oncall(incident)
