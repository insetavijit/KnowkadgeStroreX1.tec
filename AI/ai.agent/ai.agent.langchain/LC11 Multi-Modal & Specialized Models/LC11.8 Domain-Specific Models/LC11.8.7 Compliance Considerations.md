## 2. Healthcare Compliance (HIPAA)

**Protected Health Information (PHI)**:
- Names, addresses, dates
- Medical record numbers
- Social security numbers
- Any unique identifier

```python
class HIPAACompliantPipeline:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o")
        self.phi_detector = PHIDetector()  # e.g., Presidio
    
    def process(self, text: str) -> str:
        # Step 1: Check for PHI
        phi_found = self.phi_detector.analyze(text)
        
        if phi_found:
            # Option A: De-identify
            clean_text = self.phi_detector.anonymize(text)
            
            # Option B: Use on-premise model
            # response = self.on_prem_llm.invoke(text)
        else:
            clean_text = text
        
        # Process with external LLM
        response = self.llm.invoke(clean_text)
        
        # Log for audit
        self.audit_log.record({
            "timestamp": datetime.now(),
            "phi_detected": bool(phi_found),
            "action": "processed"
        })
        
        return response

# Business Associate Agreement (BAA) required with cloud providers
```

## 3. Financial Compliance

```python
class FinancialCompliancePipeline:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o")
        self.audit_log = AuditLogger()
    
    def analyze_transaction(self, transaction: dict) -> dict:
        # Explainability required
        analysis = self.llm.invoke(f"Analyze: {transaction}")
        
        # Document reasoning
        result = {
            "decision": analysis.content,
            "factors_considered": self.extract_factors(analysis),
            "model_version": "gpt-4o-2024-01",
            "timestamp": datetime.now().isoformat(),
            "analyst_review_required": True  # Human in the loop
        }
        
        # Immutable audit log
        self.audit_log.record(result)
        
        return result
    
    def check_fair_lending(self, decision: dict) -> dict:
        """Ensure no discrimination in lending decisions."""
        # Check for protected class mentions
        protected = ["race", "gender", "religion", "national origin"]
        
        for factor in decision["factors_considered"]:
            if any(p in factor.lower() for p in protected):
                raise ComplianceError(f"Protected class in factors: {factor}")
        
        return {"compliant": True}
```

## 4. Data Protection (GDPR/CCPA)

```python
class PrivacyCompliantAI:
    """GDPR/CCPA compliant AI processing."""
    
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o")
        self.consent_manager = ConsentManager()
    
    def process_user_data(self, user_id: str, data: str) -> str:
        # Check consent
        if not self.consent_manager.has_consent(user_id, "ai_processing"):
            raise ConsentError("User has not consented to AI processing")
        
        # Minimize data
        minimal_data = self.minimize_data(data)
        
        # Process
        result = self.llm.invoke(minimal_data)
        
        # Respect retention policy
        self.schedule_deletion(user_id, days=30)
        
        return result
    
    def handle_deletion_request(self, user_id: str):
        """Right to be forgotten."""
        # Delete user data from vector stores
        self.vectorstore.delete(filter={"user_id": user_id})
        
        # Delete from logs (if feasible)
        self.audit_log.anonymize(user_id)
        
        return {"deleted": True}
    
    def provide_data_export(self, user_id: str) -> dict:
        """Right to data portability."""
        return {
            "personal_data": self.collect_user_data(user_id),
            "ai_interactions": self.get_interaction_history(user_id),
            "format": "JSON"
        }
```

## 5. Audit and Documentation

```python
class AIGovernance:
    """Documentation and audit requirements."""
    
    def create_model_card(self, model_name: str) -> dict:
        return {
            "model_name": model_name,
            "version": "1.0.0",
            "intended_use": "...",
            "limitations": "...",
            "training_data": "...",
            "bias_evaluation": "...",
            "performance_metrics": {...},
            "last_updated": datetime.now().isoformat()
        }
    
    def log_decision(self, decision: dict):
        """Immutable audit log for AI decisions."""
        log_entry = {
            "id": uuid4(),
            "timestamp": datetime.now().isoformat(),
            "input_hash": hash_input(decision["input"]),
            "output": decision["output"],
            "model_version": decision["model_version"],
            "explainability": decision.get("reasoning"),
            "human_review": decision.get("reviewed_by")
        }
        
        # Append-only storage
        self.audit_store.append(log_entry)
    
    def generate_compliance_report(self, period: str) -> dict:
        """Generate compliance report for period."""
        return {
            "period": period,
            "total_decisions": self.count_decisions(period),
            "human_reviewed": self.count_reviewed(period),
            "issues_flagged": self.count_issues(period),
            "data_deletions": self.count_deletions(period)
        }
```
