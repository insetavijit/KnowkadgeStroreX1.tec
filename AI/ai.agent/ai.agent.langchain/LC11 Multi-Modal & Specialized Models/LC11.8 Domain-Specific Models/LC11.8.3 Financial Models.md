## 3. Financial Statement Analysis

```python
class FinancialAnalyst:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o")
    
    def analyze_statements(self, income: str, balance: str, cash_flow: str) -> dict:
        prompt = f"""
Analyze these financial statements:

INCOME STATEMENT:
{income}

BALANCE SHEET:
{balance}

CASH FLOW:
{cash_flow}

Provide:
1. Key Financial Ratios (calculate them)
2. Profitability Analysis
3. Liquidity Position
4. Leverage Assessment
5. Cash Flow Health
6. Red Flags or Concerns
7. Peer Comparison Context
"""
        return self.llm.invoke(prompt).content
    
    def calculate_ratios(self, data: dict) -> dict:
        """Extract and calculate key ratios."""
        prompt = f"""
From this financial data: {data}

Calculate:
- Current Ratio
- Quick Ratio
- Debt/Equity
- ROE, ROA
- Profit Margins
- P/E Ratio (if market data available)

Show formulas and results.
"""
        return self.llm.invoke(prompt).content
```

## 5. Risk and Compliance

```python
class ComplianceMonitor:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o")
    
    def check_transaction(self, transaction: dict) -> dict:
        """Flag potential compliance issues."""
        prompt = f"""
Review this transaction for compliance:
{transaction}

Check for:
1. Unusual patterns (amount, frequency, timing)
2. Sanctions list matches
3. KYC/AML concerns
4. Regulatory reporting requirements

Response: flag (yes/no), issues, recommended actions
"""
        return self.llm.invoke(prompt).content
    
    def monitor_regulatory_changes(self, regulation_text: str) -> dict:
        """Analyze new regulations for impact."""
        prompt = f"""
Analyze this regulatory update:
{regulation_text}

Identify:
1. Key requirements
2. Affected business areas
3. Compliance deadlines
4. Action items
"""
        return self.llm.invoke(prompt).content
```
